# These assume the keyring package was built prior to setting up the etc/apt/sources.list

#debian sid:
sudo sbuild-shell source:sid-riscv64-sbuild

cat /etc/apt/sources.list ; \
echo "deb http://proxy.gfnd.rcn-ee.org:3142/deb.debian.org/debian sid main" > /etc/apt/sources.list ; \
echo "deb-src http://proxy.gfnd.rcn-ee.org:3142/deb.debian.org/debian sid main" >> /etc/apt/sources.list ; \
echo "deb [trusted=yes] http://repos.rcn-ee.net/debian-riscv64 sid main" >> /etc/apt/sources.list ; \
echo "deb-src [trusted=yes] http://repos.rcn-ee.net/debian-riscv64 sid main" >> /etc/apt/sources.list ; \
cat /etc/apt/sources.list ; \
apt update ; apt install -yq bbb.io-keyring ; \
echo "deb http://proxy.gfnd.rcn-ee.org:3142/deb.debian.org/debian sid main" > /etc/apt/sources.list ; \
echo "deb-src http://proxy.gfnd.rcn-ee.org:3142/deb.debian.org/debian sid main" >> /etc/apt/sources.list ; \
echo "deb [arch=riscv64 signed-by=/usr/share/keyrings/rcn-ee-archive-keyring.gpg] http://repos.rcn-ee.net/debian-riscv64 sid main" >> /etc/apt/sources.list ; \
echo "deb-src [arch=riscv64 signed-by=/usr/share/keyrings/rcn-ee-archive-keyring.gpg] http://repos.rcn-ee.net/debian-riscv64 sid main" >> /etc/apt/sources.list ; \
cat /etc/apt/sources.list ; \
apt update ; apt clean ; echo "done..." ; exit
